4. [Anthropic/Claude] (Calvin) Q: How can I create a unit test that immitates an HTTP response in Python? This is for my function that returns something I scraped on the web.

A: Use the patch decorator and the Mock function from the unittest.mock package. Example:

import unittest
from unittest.mock import patch, Mock
import requests

def scrape_website(url):
    response = requests.get(url)
    response.raise_for_status()
    return response.json()['data']

class TestWebScraper(unittest.TestCase):
    @patch('requests.get')
    def test_scrape_website_success(self, mock_get):
        # Create a mock response object
        mock_response = Mock()
        mock_response.status_code = 200
        mock_response.json.return_value = {'data': 'scraped content'}
        mock_get.return_value = mock_response
        
        result = scrape_website('https://example.com')
        
        self.assertEqual(result, 'scraped content')
        mock_get.assert_called_once_with('https://example.com')
